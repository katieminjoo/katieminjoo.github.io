blocked Gibbs sampler

Construct a block Gibbs sampler of a linear regression model that has a normal prior.

2020-03-23

A linear regression model with a normal prior

\[y_i = X_i\beta+\epsilon_i\;\;\;\;\;\;i = {1, 2, ... N},\;\;\beta = \{\beta_1, \beta_2,...,\beta_p\}^T\] with \(\;\;\beta \sim N(0,\tau I)\;\;\) and \(\;\;\epsilon \sim N(0,\kappa I)\).

To avoid deciding \(\tau\) and \(\kappa\), we can assign hyperprior distribution to it, \(\tau \sim IG(a,b)\) and \(\;\kappa \sim IG(c,d)\)


Step 1. Find a proportionality

\[p(\beta,\tau,\kappa|Y) \propto p(Y|\beta,\kappa)\;p(\beta | \tau)\;p(\tau)\;p(\kappa)\]


each terms are \[\begin{align*} p(Y|\beta,\kappa) & = \prod_{i=1}^{N} \frac{1}{\sqrt{2\pi\kappa}} exp(-\frac{1}{2\kappa}(y_i-X_i\beta)^2)\\&= {(2\pi\kappa)}^{(-N/2)} exp(-\frac{1}{2\kappa}(Y-X\beta)^T(Y-X\beta)) \end{align*}\]

where \(\;Y\;\) is a set of \(\;y_i\;\) and \(\;X\;\) is a set of \(\;X_i\).


\[\begin{align*} p(\beta | \tau) & = \frac{1}{\sqrt{(2\pi)^P |\tau I|}} exp(-\frac{1}{2}\beta^T(\tau I)^{-1}\beta)\\&= ((2\pi)^P |\tau I|)^{-1/2} exp(-\frac{1}{2\tau}\beta^T\beta)\\&= ((2\pi)^P det(\tau I))^{-1/2} exp(-\frac{1}{2\tau}\beta^T\beta)\\&= ((2\pi)^P(\tau)^P det(I))^{-1/2} exp(-\frac{1}{2\tau}\beta^T\beta)\\&= ((2\pi)^P(\tau)^P)^{-1/2} exp(-\frac{1}{2\tau}\beta^T\beta) \end{align*}\]


\[\begin{align*} p(\tau) & = \frac{b^a}{\Gamma(a)} (\tau)^{-a-1}exp(-\frac{b}{\tau}) \end{align*}\]


\[\begin{align*} p(\kappa) & = \frac{d^c}{\Gamma(c)} (\kappa)^{-c-1}exp(-\frac{d}{\kappa}) \end{align*}\]


Step 2. Find each full conditional distribution

\[\begin{align*} p(\kappa | \tau, \beta, Y) & \propto {(2\pi\kappa)}^{(-N/2)} exp(-\frac{1}{2\kappa}(Y-X\beta)^T(Y-X\beta)) \times \kappa^{-c-1}exp(-\frac{d}{\kappa})\\& \propto {\kappa}^{-\frac{N}{2}-c-1} exp(-\frac{1}{2\kappa}(Y-X\beta)^T(Y-X\beta)-\frac{d}{\kappa})\\& \propto {\kappa}^{-(\frac{N}{2}+c)-1} exp(-\frac{1}{\kappa}\{\frac{1}{2}(Y-X\beta)^T(Y-X\beta)+d\})\\& \sim IG(c+\frac{N}{2},\; \frac{1}{2}(Y-X\beta)^T(Y-X\beta)+d) \end{align*}\]


\[\begin{align*} p(\tau | \beta, \kappa, Y) & \propto \tau^{-\frac{p}{2}}exp(-\frac{1}{2\tau}\beta^T\beta) \times \tau^{-a-1}exp(-\frac{b}{\tau})\\& = \tau^{-\frac{p}{2}-a-1}exp(-\frac{1}{2\tau}\beta^T\beta-\frac{b}{\tau})\\& = \tau^{-(\frac{p}{2}+a)-1}exp(-\frac{1}{\tau}(\frac{1}{2}\beta^T\beta+b))\\& \propto IG(\frac{p}{2}+a, \;\frac{1}{2}\beta^T\beta+b) \end{align*}\]


\[\begin{align*} p(\beta | \tau, \kappa, Y) & \propto exp(-\frac{1}{2\kappa}(Y-X\beta)^T(Y-X\beta)-\frac{1}{2\tau}\beta^T\beta)\\& = exp(-\frac{1}{2\kappa}(Y^TY-2\beta^TX^TY+\beta^TX^TX\beta)-\frac{1}{2\tau}\beta^T\beta)\\& \propto exp(-\frac{1}{2\kappa}(\beta^TX^TX\beta-2\beta^TX^TY)-\frac{1}{2\tau}\beta^T\beta)\\& = exp(-\frac{1}{2}[\kappa^{-1}\beta^TX^TX\beta+\tau^{-1}\beta^T\beta-2\kappa^{-1}\beta^TX^TY])\\& = exp(-\frac{1}{2}[\beta^T(\kappa^{-1}X^TX+\tau^{-1}I)\beta-2\kappa^{-1}\beta^TX^TY]) \end{align*}\]

\[\begin{align*} f(x) & \propto exp(-\frac{1}{2}(x-\mu)^T\Sigma^{-1}(x-\mu))\\& = exp(-\frac{1}{2}(x^T\Sigma^{-1}x-x^T\Sigma^{-1}\mu-\mu^T\Sigma^{-1}x+\mu^T\Sigma^{-1}\mu))\\& \propto exp(-\frac{1}{2}(x^T\Sigma^{-1}x-x^T\Sigma^{-1}\mu-\mu^T\Sigma^{-1}x))\\& = exp(-\frac{1}{2}(x^T\Sigma^{-1}x-2x^T\Sigma^{-1}\mu)) \end{align*}\]

\[\begin{align*} p(\beta | \tau, \kappa, Y) & \propto exp(-\frac{1}{2}[\beta^T(\kappa^{-1}X^TX+\tau^{-1}I)\beta-2\kappa^{-1}\beta^TX^TY])\\& \propto N(\frac{1}{\kappa}(\kappa X^TX+\tau^{-1}I)^{-1}X^TY, \;(\kappa X^TX+\tau^{-1}I)^{-1}) \end{align*}\]


Step 3. Iteratively update and draw new samples with data

#install.packages("invgamma")
library(invgamma)
library(MASS)

rm(list = ls())
data <- read.csv("regression_data.csv")
Y <- data$Y
X <- as.matrix(data[,2:4])
N <- dim(X)[1]
p <- dim(X)[2]

tau_samples <- c()
beta_samples <- list()
kappa_samples <- c()
# ======= Hyperparameters ======= #
a <- 1 ; b <- 1 
c <- 1 ; d <- 1
num_iters <- 10000
# ======= initial values ======= # 
beta <- rep(0,p)
for(i in 1:num_iters){
  # ======= update kappa ======= #
  kappa <- rinvgamma(1, c+N/2, t(Y - X%*%beta) %*% (Y - X%*%beta)/2 + d)
  
  # ======= update tau ======= #
  tau <- rinvgamma(1, a+p/2, t(beta)%*%beta/2+b)
  
  # ======= beta ====== #
  cov_beta <- solve(t(X)%*%X/kappa + diag(p)/tau)
  mu_beta <- cov_beta %*% t(X) %*% Y/kappa
  beta <- mvrnorm(1, mu=mu_beta, Sigma=cov_beta)
  
  # ======= collect samples ======= #
  kappa_samples[i] <- kappa
  tau_samples[i] <- tau
  beta_samples[[i]] <- beta
}
